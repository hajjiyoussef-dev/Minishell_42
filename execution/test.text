NAME = libexecution.a

SRC = ./parsing_test/lexer.c ./parsing_test/shell.c  execution.c gc_malloc.c \
			execution_tools.c ft_split.c ft_strjoin.c ./parsing_test/syntax.c ./parsing_test/expand.c \

INC = ../test.h
